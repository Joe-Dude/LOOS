<body>
    <div class="browser-header">
        <div class="search-bar">
            <input type="text" id="urlInput" placeholder="Enter URL or search term..." autocomplete="off" autocorrect="off" autocapitalize="off" spellcheck="false">
            <button onclick="navigate()">Go</button>
        </div>
    </div>
    
    <div class="error-message" id="errorMessage">
        Note: Some websites prevent being loaded in iframes for security reasons.
    </div>
    
    <div class="content-area">
        <div class="home-page" id="homePage">
            <h1>Iframe Browser</h1>
            <p>Enter a URL or search term to get started</p>
            <div class="home-search">
                <input type="text" id="homeSearch" placeholder="Search or enter URL..." autocomplete="off" autocorrect="off" autocapitalize="off" spellcheck="false">
                <button onclick="searchFromHome()">Search</button>
            </div>
        </div>
        <iframe id="browser"></iframe>
    </div>

    <script>
        const urlInput = document.getElementById('urlInput');
        const homeSearch = document.getElementById('homeSearch');
        const browserFrame = document.getElementById('browser');
        const homePage = document.getElementById('homePage');
        const errorMessage = document.getElementById('errorMessage');
        
        function searchFromHome() {
            const query = homeSearch.value.trim();
            if (query) {
                urlInput.value = query;
                navigate();
            }
        }
        
        function navigate() {
            let url = urlInput.value.trim();
            
            if (!url) {
                showHome();
                return;
            }
            
            // If it doesn't start with http:// or https://, add https://
            if (!url.startsWith('http://') && !url.startsWith('https://')) {
                url = 'https://' + url;
            }
            
            urlInput.value = url;
            browserFrame.src = url;
            homePage.classList.add('hidden');
            browserFrame.classList.add('active');
            errorMessage.style.display = 'none';
        }
        
        function showHome() {
            homePage.classList.remove('hidden');
            browserFrame.classList.remove('active');
            browserFrame.src = '';
            urlInput.value = '';
            homeSearch.value = '';
        }
        
        // Allow Enter key to navigate
        urlInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                navigate();
            }
        });
        
        homeSearch.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                searchFromHome();
            }
        });
        
        // Show error message if iframe fails to load
        browserFrame.addEventListener('error', function() {
            errorMessage.style.display = 'block';
        });
    </script>
